unittestresults: clean unittests cardtests
	@echo "Results for Assignment3 unit testing:" > tests.out
		
	@echo "unittest1:" | tee -a tests.out
	@/bin/rm -f dominion.gcda
	@ ./unittest1 | tee -a tests.out
	@gcov dominion.c | tee -a tests.out

///////////////////////////////////////////////////////////////////
	
CFLAGS = -Wall -fpic -coverage -lm -std=c99

default: unittestresults.out

unittestresults.out: unittest1.c unittest2.c unittest3.c unittest4.c dominion.c dominion.h dominion_helpers.h rngs.h
	
	echo "Result for running tests:" > unittestresults.out
	echo "unittest1.c:" >> unittestresults.out
	gcc -o unittest1 dominion.c dominion.h dominion_helpers.h rngs.h unittest1.c $(CFLAGS)
	./unittest1 >> unittestresults.out
	gcov dominion.c dominion.h dominion_helpers.h rngs.h >> unittestresults.out
	
	echo "Result for running tests:" > unittestresults.out
	echo "unittest2.c:" >> unittestresults.out
	gcc -o unittest2 dominion.c dominion.h dominion_helpers.h rngs.h unittest2.c $(CFLAGS)
	./unittest2 >> unittestresults.out
	gcov dominion.c dominion.h dominion_helpers.h rngs.h >> unittestresults.out
	
	echo "Result for running tests:" > unittestresults.out
	echo "unittest3.c:" >> unittestresults.out
	gcc -o unittest3 dominion.c dominion.h dominion_helpers.h rngs.h unittest3.c $(CFLAGS)
	./unittest3 >> unittestresults.out
	gcov dominion.c dominion.h dominion_helpers.h rngs.h >> unittestresults.out
	
	echo "Result for running tests:" > unittestresults.out
	echo "unittest4.c:" >> unittestresults.out
	gcc -o unittest4 dominion.c dominion.h dominion_helpers.h rngs.h unittest4.c $(CFLAGS)
	./unittest4 >> unittestresults.out
	gcov dominion.c dominion.h dominion_helpers.h rngs.h >> unittestresults.out

clean:
	rm unittestresults.out

cleanall: 
	rm unittestresults.out unittest1 unittest2 unittest3 unittest4 *.gcda *.gcno *.gcov
